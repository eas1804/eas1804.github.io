cat docker-compose.yaml

---
services:
  adguardhome:
    image: adguard/adguardhome
    container_name: adguardhome
    restart: unless-stopped
    volumes:
      - ./workdir:/opt/adguardhome/work
      - ./confdir:/opt/adguardhome/conf
    ports:
      - "53:53/tcp"
      - "53:53/udp"

      - "3000:3000/tcp"
      - "80:80/tcp"

      - "443:443/tcp"
      - "443:443/udp"

      - "67:67/udp"
      - "68:68/udp"


      - "853:853/tcp"
      - "784:784/udp"
      - "853:853/udp"
      - "8853:8853/udp"
      - "5443:5443/tcp"
      - "5443:5443/udp"
  1. port 3000/tcp - Панель установки
  2. port 80,443/tcp - Админпанель
  3. port 67:67/udp, 68:68/tcp, 68:68/udp - использовать AdGuard Home в качестве DHCP-сервера.
  4. port 853:853/tcp - запустить AdGuard Home как сервер DNS-over-TLS⁠.
  5. port 784:784/udp, 853:853/udp, 8853:8853/udp - запустить AdGuard Home как DNS-over-QUIC⁠ сервер.
  6. port 5443:5443/tcp, 5443:5443/udp - запустить AdGuard Home как DNSCrypt⁠ server.
