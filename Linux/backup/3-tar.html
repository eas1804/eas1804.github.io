tar

Сделать копию каталога /mnt/base_1c/ в /backup/1c/ 
#!/bin/sh

TO=/backup/1c/		# Папка, куда будем складывать архивы. 	/в конце - обязательна
FROM=/mnt/base_1c/  # Папка, которую будем архивирвоать. 	/в конце - обязательна

tar -vczf $TO`date "+%d-%m-%g_%H-%M-%S"`-1с.tar.gz   $FROM
exit0
Если каталогов много, можно использовать функцию 
#!/bin/sh

backup_func (){
tar -vczf $TO`date "+%d-%m-%g_%H-%M-%S"`-User.tar.gz   $FROM
}

# Folder1
TO=/backup/Ivanoff/			# Папка, куда будем складывать архивы. 	/в конце - обязательна
FROM=/mnt/Ivanoff/  		# Папка, которую будем архивирвоать. 	/в конце - обязательна
backup_func

# Folder2
