#Проверяем наличие каталогов increment и current на удаленном сервере
ssh -p$ssh_port -i $ssh_key $username@$server ls ${target_directory}increment/ || ssh -p$ssh_port -i $ssh_key $username@$server mkdir -p ${target_directory}increment/
ssh -p$ssh_port -i $ssh_key $username@$server ls ${target_directory}current/   || ssh -p$ssh_port -i $ssh_key $username@$server mkdir -p ${target_directory}current/

# Синхронизируем с удаленным сервером
rsync -zav --progress --delete -e "ssh -p$ssh_port -i $ssh_key" --recursive $local_directory $username@$server:$target_directory/current/ \
 --backup --backup-dir=${target_directory}/increment/`date +%Y-%m-%d`/

# Удаляем старые каталоги в increment
ssh -p$ssh_port -i $ssh_key $username@$server ls -dt "${target_directory}increment"/*/ | tail -n +$((COUNT+1)) | \
xargs -I {} ssh -p$ssh_port -i $ssh_key $username@$server  rm -r -- "{}"

# Есть ли ошибки?
if [ $? -ne 0 ]; then
MESSAGE="Ошибка выполнения скрипта
$echo $0
LAN ip $IP_local
WAN ip $WAN $IP_global
$local_directory

Remote:
$server
$target_directory
"

else
    echo "Команда выполнена успешно."
fi
}
